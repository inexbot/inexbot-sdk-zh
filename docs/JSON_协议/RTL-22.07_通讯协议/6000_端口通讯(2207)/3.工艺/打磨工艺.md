# 打磨工艺

#### 打磨工艺参数设置

*0x4601 POLISH_PARAM_SET

{

"robot":1,

"craftID":1,

"polish":

{

"toolWear":0.0,         //工具磨损补偿

"startToolOffset":

{

"X":0.0,          //起点工具X方向补偿

"Y":0.0,

"Z":0.0

},

"endToolOffset":

{

"X":0.0,          //终点工具X方向补偿

"Y":0.0,

"Z":0.0

},

"autoFeed":

{

"value":0.0,          //自动补偿值

"cycle":0          //自动补偿周期（次）

}

}

}

#### 查询打磨工艺参数

*0x4602 POLISH_PARAM_INQUIRE

{

"robot":1,

"craftID":1   // 1 - 9

}

#### 返回0x4602查询结果

*0x4603 POLISH_PARAM_RESPOND

{

"robot":1,

"craftID":1,

"polish":

{

"toolWear":0.0,

"startToolOffset":

{

"X":0.0,

"Z":0.0

},

"endToolOffset":

{

"X":0.0,

"Y":0.0,

"Z":0.0

},

"autoFeed":

{

"value":0.0,

"cycle":0.0

}

}

}
