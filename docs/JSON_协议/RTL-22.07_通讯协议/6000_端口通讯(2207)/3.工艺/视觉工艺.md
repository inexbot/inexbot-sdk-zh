#### 视觉位置参数设置​
​*0x4104 VISION_POS_PARAMETER_SET

{

"position": {

"angleDirection": 1,  //角度方向 1或-1

"cameraPoint": [0.0, 0.0, 100.0, 0.0, 0.0, 0.0],  //第一位是X值，第二位是Y值，第三位是高度，第四位是角度;相机点位：若参数类型是2D和2D+H，则保存4元数组[x,y,H,angle];若为3D，则保存6元数组[x,y,z,a,b,c]。

"datumPoint": [0.0, 0.0, 0.0, 3.140, 0.0, 0.0, 0.0],  //基准点x,y,z,a,b,c

"excursion": {

"Xexcursion": 1.0,   //X方向偏移

"Yexcursion": 2.0,  //Y方向偏移

"Zexcursion": 3.0,  //Z方向偏移

"angle": 4.0        //角度偏移

},

"recvPointsType": 0,   //接受位置类型：0：点，1：轨迹

"scale": 1.0   //比例系数

},

"robot": 1,  //机器人

"visionNum": 0   //工艺号从0开始

}

{

"robot":1,			//1~4

"visionNum":0,		//0~8

"visionRange":

"maxX":"0",		//将数字转换为字符串发送，如果未设置，则发送空字符串""，

"maxY":"1.1",

"maxZ":"222",

"minX":"0.003",

"minY":"0",

"minZ":"0"

}

}
