# 3D鼠标控制

#### 设置灵敏度发送

*0x7301 THREED_MOUSE_SET

{

"ABC":4 or 5 or 6   //  姿态ABC

"mouseSen":[128,128,128,128]  //鼠标灵敏度xyz和姿态灵敏度（0-300）

}

#### 获取

*0x7302 THREED_MOUSE_INQUIRE

data:无

#### 控制器回复：

*0x7303 THREED_MOUSE_RESPOND

{

//"enable":false

"ABC":4 or 5 or 6   //  姿态ABC

"mouseSen":[128,128,128,128]   //鼠标灵敏度xyz和姿态灵敏度（0-300）

}

#### 3D鼠标零点标记零点发送

*0x7304 THREED_MOUSE_SETZERO

data:无

#### 获取零点标记状态

*0x7305 THREED_MOUSE_SETZERO_INQUIRE

data:无

#### 控制器回复：

*0x7306 THREED_MOUSE_SETZERO_RESPOND

{

"zeroTagged":0 1 2   //0：未标记，1：已标记，2：通讯失败

}

#### 3D鼠标正方向标记设置：

*0x7307 THREED_MOUSE_SIGN_DIRECTION

{

"axis":1 or 2 or 3   //1表示标记X正方向，2：Y，3：Z

}

#### 获取3D鼠标正方向标记状态

*0x7308 THREED_MOUSE_SIGN_DIRECTION_INQUIRE

data:无

#### 获取3D鼠标正方向标记状态，控制器回复：

*0x7309 THREED_MOUSE_SIGN_DIRECTION_RESPOND

{

"xTagged":0 1 2   //X正方向0：失败，1：成功，2：标定中

"yTagged":0 1 2   //Y正方向0：失败，1：成功，2：标定中

"zTagged":0 1 2   //Z正方向0：失败，1：成功，2：标定中

}

#### 退出页面时示教器发送：

*0x730A PAGE_BACK

{

"pageBack":1 //3D鼠标点标记方向但未标记，直接切换界面时通知控制器退出标记方向线程

}

#### 设置3D鼠标端口号

*0x730B THREED_MOUSE_SET_PORT

{

"port" : 1,    //1-10

}

#### 查询3D鼠标端口号

*0x730C THREED_MOUSE_INQUIRE_PORT

{

}

#### 返回0x730C查询结果

*0x730D THREED_MOUSE_RESPOND_PORT

{

"port" : 1,    //1-10

}
