# 数据上传

#### 设置数据上传开关 发送：

*0x5721    DATAUPLOAD_ENABLE_SET

{

"robot":1,    //机器人号

"enable":true/false;   //数据是否上传

}

#### 查询数据传输状态

*0x5722   DATAUPLOAD_ENABLE_INQUIRE

{

"robot":1,   //机器人号

}

#### 返回数据传输状态

*0x5723   DATAUPLOAD_ENABLE_RESPOND

{

"robot":1,    //机器人号

"enable":true/false;    //数据是否上传

}

#### 设置数据传输数据

*0x5724     DATAUPLOAD_PARM_SET// 注释不全

{

"robot":1,     //机器人号

"uploadMode":  			//string 上传方式

"fileSuffix":				//string 文件格式

"serverIP":				//string 服务器IP

"serverPort":				//int 端口号

"username":				//string 用户名

"password":				//string密码

"filepath":				//string 路径

"dataCollectTime":			//double数据采集周期

"dataUploadTime":		//double数据上传周期

"sendInfoFileFlag":			//bool是否发送说明文件

}

#### 查询数据传输数据

*0x5725   DATAUPLOAD_PARM_INQUIRE

{

"robot":1,   //机器人号

}

#### 返回数据传输数据

*0x5726   DATAUPLOAD_PARM_RESPOND

data  同0x5724

#### 设置上传数据格式

*0x5727   DATAUPLOAD_FORMAT_SET

{

"robot":1,   //机器人号

"type":1,//1,2,3,4

"format":""   //说明文档文件名

}

#### 查询数据格式

*0x5728  DATAUPLOAD_FORMAT_INQUIRE

{

"robot":1,

"type":1,//1,2,3,4

}

#### 返回数据格式

*0x5729   DATAUPLOAD_FORMAT_RESPOND

{

"robot":1,   //机器人号

"type":1,//1,2,3,4

"format":""   //说明文档文件名

}
