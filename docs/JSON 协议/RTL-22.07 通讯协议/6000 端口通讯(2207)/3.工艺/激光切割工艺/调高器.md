# 调高器

====================================================================

0x91A1  跟随开命令请求

{

"robot":1,

"thirdLevelPerforationEnable":false,  //三级穿孔使能

"secondPerforationEnable":false,  //二次穿孔使能

"perforationMode":1  //0：直接穿孔使能 1：分段穿孔使能 2：渐近穿孔使能

}

=========================================================================

0x91A2  跟随关命令请求

{

"robot":1,

}

=========================================================================

0x91A3  跟随停命令请求

{

"robot":1,

}

=========================================================================

0x91A4  调高器点动

{

"robot":1,

"direction":1, // 0:下降 1:上升    (100ms内发送)

"speedLevel":1, // 0:低速 1:中速 2:高速

}

0x91AF  停止点动

{

"robot":1,

}

============STATUS值枚举===============================

enum class OperationStatus{

STOP_ = 0x00,              // 停止

MENU_SETTING_ = 0x01,      // 菜单设定中

MOTOR_CALIBRATION_ = 0x02, // 电机标定中

CAPACITANCE_CALIBRATION_ = 0x03, // 电容标定中

AUTO_TUNING_ = 0x04,       // 自动整定中

HOME_ = 0x05,              // 回零

FOLLOW_ = 0x06,            // 跟随

ASCEND_ = 0x07,            // 上升

DESCEND_ = 0x08,           // 下降

EMERGENCY_TOP_ = 0x09,    // 急停

FROG_LEAP_ = 0x0A          // 蛙跳

};

0x91A7  请求调高器相关信息

{

"robot":1

}

0x91A8  返回调高器相关信息

{

"status":  0, // OperationStatus

"capacitance_value":0, //电容值

"position":0, //位置值

"height":0, //高度值

"version":0,

"stop_coordinate":1,

"air_moving_board_delay":1,

"back_to_center_coordinate"1,

"cutting_plate_delay":1,

"perforation_plate_delay":1,

"zaxis_coordinate":1,

}

0x91A9  设置跟随等级

{

”cuttingTrackLevel“：30,  //范围(0,30]

"rAngleTrackLevel":0

}
