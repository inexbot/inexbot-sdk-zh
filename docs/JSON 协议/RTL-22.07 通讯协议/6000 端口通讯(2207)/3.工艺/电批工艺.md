# 电批工艺

#### 修改电批参数

*0x5731   SCREWDRIVER_PARM_SET

```json
{
"proNum":1//int型
"torqueCorrection":0，      //扭力补正值:-10-10
"tighten":      //拧紧
{
"rotationDir":0,        //旋转方向：0：CW，1：CCW
"targetTorque":1.1,         //目标扭力，单位mN*m
"rounds":[1,2,3,4,5,6,7,8,9],         //步骤0-9的圈数：0-20.0，double型
"speed":[1,2,3,4,5,6,7,8,9],          //步骤0-9的速度：10-1500，double型
"tightenMode":0,             //拧紧模式：0：标准，1：攻丝，2：高速
"initDetectionAmountFlag":true,         //是否开启初始检出量
"initDetectionAmount":0,          //初始检出量（mN*m）：0-10，double型
"initTapTorqueFlag":true，        //是否开启初始攻丝扭力
"initTapTorque":0，        //初始攻丝扭力（mN*m）：10-700，double型
"torqueHoldTimeFlag":true，      //是否开启扭力保持时间
"torqueHoldTime":0，        //扭力保持时间（sec）,double型
"antiLimitFlag":true，        //是否开启-侧许可圈数
"antiLimit":-10，            //-侧许可圈数：-10-0，double型
"positiveLimitFlag":true,         //是否开启+侧许可圈数
"positiveLimit":10,            //+侧许可圈数：0-10，double型
"addTightenAngleFlag":true,          //是否开启增加拧紧角度（°）
"addTightenAngle":360，            //增加拧紧角度：-360-360
"floatTightenAngleFlag":true,            //是否开启浮高后拧紧角度(°)
"floatTightenAngle":360,                     //浮高后拧紧角度(°)：0-360，double型
"velSwitchTorqueRatioFlag":true,         //是否开启速度切换扭力
"velSwitchTorqueRatio":1,                  //速度切换扭力比值（%），double型
"speedAfterSwitchFlag":true,        //是否开启切换后速度
"speedAfterSwitch":10            //切换后速度(rpm)：10-1000，double型
}
"loosen":        //拧松
{
"rounds":[1.0 , 2.0 , 3.0],    //步骤0-1圈数，-20.0-20.0
"speed":[1.0 , 2.0 , 3.0],      //步骤0-1速度,10-1000
"rotationDir":0,        //旋转方向：0：CW，1：CCW
}
}
```
